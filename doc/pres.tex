\section{Preface}

\frame
{
 \frametitle{Goal}
 \begin{block}{Goal}
   Generating a despiriation depth map of the environment using stereo vision.
 \end{block}
}

\frame{
 \frametitle{Why it is interesting}
   A depthmap can be used for various purposes:
   \begin{itemize}
    \item 3D modeling of a 2D enviroment
    \item Tracking of objects
    \item Recognising front objects
    \item As information about the enviroment in path planning.
   \end{itemize}
}


\section{Problems}

\frame{
 \frametitle{Theoretical problems}
   Stereo vision in a real live enviroment can be split up in serveral subproblems:
   \begin{itemize}
    \item Camera callibration problems
    \item Uncertain epipolair line
    \item Occlusion
    \item Matching object in both images
   \end{itemize}
}

\section{Approach}

\frame
{
  \frametitle{Approach}
  \begin{itemize}
    \item Camera calibration
    \item Epipolar geometry
     \item Dense stereo algorithms
      \begin{itemize}
       \item Graph Cut
       \item Believe Propagation
       \item Region Based
      \end{itemize}
    \item Using the OpenCV library
  \end{itemize}
}

\frame{
  \frametitle{Separate Goals}
  The goal can be seperated into two independant subgoals:

  \begin{itemize}
   \item \textbf{Calibration and Rectification} Starting with two camera's and building a rectified image
   \item \textbf{Dense Stereo} Starting with a rectified image and bulding a dense despiration map \\ \textit{Can use an external dataset}
  \end{itemize}
}

\frame
{
  \frametitle{Practical problems}
  \begin{itemize}
   \item Getting webcams to work
   \item Learning OpenCV
   \item Selecting and understanding the right dense stereo algorithm.
  \end{itemize}
}
